\subsection*{Statistique}
% \frame{\tableofcontents[currentsubsection]}
%====================================================================
\frame{\frametitle{Modčles ŕ structure latente}

  \paragraph{Cadre général.}
  \begin{description}
  \item [$\theta =$] paramčtres (dimension modérée)
  \item [$Z =$] variables latentes (dimensions comparables aux données)
  \item [$Y =$] données observées
  \end{description}

  \bigskip \bigskip \pause
  \paragraph{Enjeu.} Vraisemblance $\displaystyle{p_\theta(Y) = \int p_\theta(Z, Y) \d Z}$ souvent incalculable
  \begin{itemize}
  \item Détour par la vraisemblance jointe (``complčte'') $p_\theta(Z, Y)$ 
  \item Détermination de \comment{(certains moments de)} la loi conditionnelle $p_\theta(Z \mid Y)$
  \end{itemize}

  \bigskip \bigskip \pause
  \paragraph{Trois approches classiques} pour $p_\theta(Z \mid Y)$: 
  $$
  \text{\emphase{($a$)} Calculer \qquad \qquad
  \emphase{($b$)} Approcher \qquad \qquad
  ($c$) Échantillonner} 
  $$
  
}

%====================================================================
\frame{\frametitle{($a$) Inférence ``exacte''}

  \paragraph{Structure latente discrčte:} $Z \in \Zcal$ avec $|\Zcal|$ fini
  \begin{itemize}
   \item Sommation sur $\Zcal$ en exploitant sa structure algébrique
  \end{itemize}
  
  \bigskip \pause
  \paragraph{Détection de ruptures.} Pour $k$ segments, $|\Zcal| = \binom{n-1}{k-1}$
  \begin{itemize}
   \item Segmentation optimale: programmation dynamique ({\sl max-sum})
   \item Intégration sur $\Zcal$: produit matriciel ({\sl sum-product})
   \item Distribution conditionnelle ``exacte'' des points de ruptures \Refer{G. Rigaill, A. Cleynen}
  \end{itemize}

  \bigskip \pause
  \paragraph{Inférence d'un modčle graphique.} $Z =$ arbre recouvrant sur $p$ noeuds, $|\Zcal| = p^{p-2}$
  \begin{itemize}
   \item Arbre optimal: algorithme de Kruskal (Chow \& Liu)
   \item Intégration sur $\Zcal$: théorčme arbre-matrice (Kirchhoff)
   \item Probabilité conditionnelle ``exacte'' des arętes \Refer{L. Schwaller, R. Momal}
  \end{itemize}

  \bigskip \pause
  \paragraph{Détection de ruptures dans un modčle graphique.} 
  \begin{itemize}
   \item Combinaison des deux \Refer{L. Schwaller}
  \end{itemize}

}

%====================================================================
\frame{\frametitle{ ($b$) Inférence ``approchée''}

  \paragraph{Approximation variationnelle:} $q(Z) \simeq p(Z \mid Y)$
  $$
  \widetilde{q} = \argmin_{q \in \emphase{\Qcal}} \emphase{D}\left[q \,||\, p(\cdot \mid Y)\right]
%   \qquad \qquad (\text{ex: } D = KL)
  $$
  $D = KL \Rightarrow$ EM variationnel (VEM): minimise une borne inférieure de la log-vraisemblance 
  
  \bigskip \bigskip \pause
  \paragraph{Modčles ŕ blocs stochastiques (SBM).} $n$ noeuds, $Y_{ij} = 1$ si $i$ et $j$ interagissent
  \begin{itemize}
  \item $\{Z_i\}$ iid: $Z_i \sim \Mcal(1, \pi)$ 
  \item $\{Y_{ij}\} \mid \{Z_i\}$ indépendants $P(Y_{ij} = 1) = \gamma_{Z_iZ_j}$
  \item approximation de champ moyen: $p(Z \mid Y) \simeq \prod_i \Mcal(Z_i; 1, \tau_i)$ 
  \end{itemize}

  \bigskip \bigskip \pause
  \paragraph{Modčle Poisson log-normal (PLN).} $n$ sites, $p$ espčces, $Y_{ij} =$ abondance de l'espčce $j$ dans le site $i$
  \begin{itemize}
  \item $\{Z_i\}$ iid: $Z_i \sim \Ncal_p(0, \Sigma)$ 
  \item $\{Y_{ij}\} \mid \{Z_i\}$ indépendants: $Y_{ij} \sim \Pcal\left(e^{x_i^\intercal \beta_j + Z_{ij}}\right)$
  \item approximation gaussienne: $p(Z_i \mid Y_i) \simeq \Ncal(Z_i; m_i, S_i)$ 
  \end{itemize}

}

\frame{\frametitle{Distribution d'espčces}

  \paragraph{Écologie des communautés.}
  \begin{itemize}
   \item {\sl Drivers} environnementaux de l'abondance de différentes espčces (effets {\sl abiotiques})
   \item Interactions entre ces męmes espčces (interactions {\sl biotiques})
  \end{itemize}
  
  \bigskip \bigskip \pause
  \paragraph{PLN = modčle joint de distributions d'espčces.} $n$ sites ($i$), $p$ espčces ($j$)
  \begin{align*}
    Z_i & \sim \Ncal_p(0, \emphase{\Sigma}) &
    \Sigma & = \text{``relations biotiques''} \\ ~ \\
    Y_{ij} & \sim \Pcal(\exp(x_i^\intercal \emphase{\beta_j} + Z_{ij})) & 
    \beta_j & = \text{``effets abiotiques''}
  \end{align*}
  \ra Inférence par VEM: package \url{PLNmodels} \Refer{J. Chiquet, M. Mariadassou}

  \bigskip \bigskip \pause
  \paragraph{Extensions et utilisation.}
  \begin{itemize}
   \item Réduction de dimension ($\Sigma$ de rang faible), inférence de réseau ($\Sigma^{-1}$ creuse), classification (mélanges, analyse discriminante), ...
   \item Formations ŕ destination d'écologues (ANR NGB et EcoNet)
   \item Plusieurs articles en collaboration
  \end{itemize}
