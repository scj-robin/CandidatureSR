\chapter{Projet}

\bigskip
\section{Enjeux pour la biologie (moléculaire)}

\subsection{Tendance}% lourde}
\vspace{-0.5cm}
\begin{itemize}
\item Les technologies modernes rendent la biologie de plus en plus
  quantitative\vspace{-0.5cm}
\item Les biologistes se trouvent confrontés ŕ des masses de données
  impossibles ŕ analyser ``ŕ la main'' et ŕ interpréter ``ŕ l'oeil''
\vspace{-0.5cm}
\item Ces données sont souvent extręmement variables
\end{itemize}

\noindent $\Rightarrow$ La modélisation (informatique, mathématique,
statistique) est incontournable pour tirer une information pertinente
des données observées, voire pour bien poser les questions
biologiques %\vspace{-0.5cm}

\noindent Mais les outils pertinents ne sont pas forcément nouveaux 

\paragraph{$\Rightarrow$ Besoin de (statisticiens) modélisateurs}
capables de 
\vspace{-0.5cm}
\begin{itemize}
\item Comprendre les besoins et les questionnements des
  biologistes\vspace{-0.5cm}
\item De proposer des premičres réponses fondées sur des
  méthodes existantes
\vspace{-0.5cm}
\item De développer de nouvelles méthodes si besoin est
\end{itemize}
